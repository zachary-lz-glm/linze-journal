## 一、JS 语言特性（5 题）

### H1. 深拷贝（处理循环引用）

```javascript
function deepClone(obj, map = new WeakMap()) {
  // 基本类型直接返回
  if (obj === null || typeof obj !== 'object') return obj;
  // 循环引用：已拷贝过直接返回
  if (map.has(obj)) return map.get(obj);
  // 数组或对象
  const clone = Array.isArray(obj) ? [] : {};
  map.set(obj, clone);
  // Reflect.ownKeys 包含 Symbol 键
  for (const key of Reflect.ownKeys(obj)) {
    clone[key] = deepClone(obj[key], map);
  }
  return clone;
}
```

**追问点：**
- 为什么用 WeakMap 不用 Map？→ WeakMap 弱引用不阻止 GC
- 能处理 Date/RegExp/Map/Set 吗？→ 需要加类型判断分支
- structuredClone API？→ 浏览器原生，但不能处理函数

### H2. 防抖和节流

```javascript
// 防抖：连续触发只执行最后一次
function debounce(fn, delay) {
  let timer;
  return function (...args) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

// 节流：固定间隔执行一次
function throttle(fn, interval) {
  let lastTime = 0;
  return function (...args) {
    const now = Date.now();
    if (now - lastTime >= interval) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}
```

**追问点：**
- 防抖要不要leading（首次立即执行）？→ 加 immediate 参数
- 节流要不要trailing（最后一次也执行）？→ 加定时器兜底
- 取消功能？→ 返回的对象加 cancel 方法

```javascript
  // 沿原型链查找
  let proto = Object.getPrototypeOf(obj);
  while (proto) {
```

### H4. 手写 call / apply / bind

```javascript
// call
Function.prototype.myCall = function (ctx, ...args) {
  ctx = ctx || window;
  const fn = Symbol(); // 防止覆盖已有属性
  ctx[fn] = this;
  const result = ctx[fn](...args);
  delete ctx[fn];
  return result;
};

// apply
Function.prototype.myApply = function (ctx, args) {
  ctx = ctx || window;
  const fn = Symbol();
  ctx[fn] = this;
  const result = ctx[fn](...args);
  delete ctx[fn];
  return result;
};

// bind（支持柯里化 + new）
Function.prototype.myBind = function (ctx, ...bindArgs) {
  const fn = this;
  const bound = function (...callArgs) {
    // new 调用时 this 指向实例，忽略 ctx
    return fn.apply(this instanceof fn ? this : ctx, [...bindArgs, ...callArgs]);
  };
  bound.prototype = Object.create(fn.prototype);
  return bound;
};
```

**追问点：**
- 为什么用 Symbol？→ 防止和对象已有属性冲突
- bind 返回的函数被 new 调用怎么办？→ this 指向实例而非 ctx
- apply 的 args 不是数组怎么办？→ 加类型判断

### H5. ES6 class 转 ES5（寄生组合继承）

```javascript
// ES6 class
class Animal {
  constructor(name) { this.name = name; }
  speak() { console.log(this.name); }
}
class Dog extends Animal {
  constructor(name) { super(name); this.type = 'dog'; }
  bark() { console.log('wang'); }
}

// ES5 等价实现
function AnimalES5(name) { this.name = name; }
AnimalES5.prototype.speak = function () { console.log(this.name); };

function DogES5(name) {
  AnimalES5.call(this, name); // super(name)
  this.type = 'dog';
}
// 寄生组合继承：不用 DogES5.prototype = new AnimalES5()
DogES5.prototype = Object.create(AnimalES5.prototype);
DogES5.prototype.constructor = DogES5;
DogES5.prototype.bark = function () { console.log('wang'); };
```

## 二、异步控制（3 题）

### H6. Promise 并发调度器（字节/快手高频）

```javascript
class Scheduler {
  constructor(max) {
    this.max = max;
    this.count = 0;
    this.queue = [];
  }
  add(promiseCreator) {
    return new Promise((resolve) => {
      if (this.count < this.max) {
        this.count++;
        promiseCreator().then(() => {
          this.count--;
          resolve();
          this.next();
        });
      } else {
        this.queue.push({ promiseCreator, resolve });
      }
    });
  }
  next() {
    if (this.queue.length && this.count < this.max) {
      const { promiseCreator, resolve } = this.queue.shift();
      this.count++;
      promiseCreator().then(() => {
        this.count--;
        resolve();
        this.next();
      });
    }
  }
}

// 使用：控制最多 2 个并发
// const s = new Scheduler(2);
// s.add(() => fetch('/api1')).then(() => console.log('done1'));
// s.add(() => fetch('/api2')).then(() => console.log('done2'));
// s.add(() => fetch('/api3')).then(() => console.log('done3'));
```

```javascript
// 串行执行
async function promiseSerial(tasks) {
  const results = [];
  for (const task of tasks) {
    results.push(await task());
  }
  return results;
}

// 带重试
function withRetry(fn, retries = 3, delay = 1000) {
  return new Promise(async (resolve, reject) => {
    for (let i = 0; i < retries; i++) {
      try {
        const result = await fn();
        return resolve(result);
      } catch (e) {
        if (i === retries - 1) return reject(e);
        await new Promise((r) => setTimeout(r, delay));
      }
    }
  });
}
```

```javascript
Promise.myAll = function (promises) {

    const results = [];
    let count = 0;
    promises.forEach((p, i) => {
      Promise.resolve(p).then(
        (val) => {
```

## 三、设计模式（2 题）

### H9. EventEmitter 发布订阅

```javascript
class EventEmitter {
  constructor() { this.events = {}; }
  on(event, fn) {
    (this.events[event] ||= []).push(fn);
    return this; // 支持链式调用
  }
  emit(event, ...args) {
    (this.events[event] || []).forEach((fn) => fn(...args));
    return this;
  }
  off(event, fn) {
    this.events[event] = (this.events[event] || []).filter((f) => f !== fn);
    return this;
  }
  once(event, fn) {
    const wrap = (...args) => {
      fn(...args);
      this.off(event, wrap);
    };
    this.on(event, wrap);
    return this;
  }
}
```

**追问点：**
- once 实现原理？→ 包装函数执行后自动 off
- 多个 once 监听同一事件？→ 每次生成新的 wrap 函数
- 和你项目里的 Listener 发布订阅的关系？→ Schema 联动引擎用的就是类似机制

### H10. 模板编译函数（拼多多高频）

```javascript
function compile(template, data) {
  // 匹配 {{ key }} 并替换
  return template.replace(/\{\{\s*(\w+)\s*\}\}/g, (match, key) => {
    return data[key] !== undefined ? data[key] : '';
  });
}

// 进阶版：支持嵌套属性 {{ user.name }}
function compileAdvanced(template, data) {
  return template.replace(/\{\{\s*([\w.]+)\s*\}\}/g, (match, path) => {
    return path.split('.').reduce((obj, key) => obj?.[key], data) ?? '';
  });
}

// 使用
// compile('你好 {{ name }}，你有 {{ count }} 条消息', { name: '张三', count: 5 })
// → '你好 张三，你有 5 条消息'
```

```javascript
function twoSum(nums, target) {
  const map = new Map();
  for (let i = 0; i < nums.length; i++) {
    if (map.has(target - nums[i])) return [map.get(target - nums[i]), i];
    map.set(nums[i], i);
  }
}
```

口诀：遍历时查 Map，查到就返回，查不到就存进去。

### H12. 双指针 + 滑动窗口

```javascript
// 对撞指针：三数之和
function threeSum(nums) {
  nums.sort((a, b) => a - b);
  const res = [];
  for (let i = 0; i < nums.length - 2; i++) {
    if (i > 0 && nums[i] === nums[i - 1]) continue; // 去重
    let l = i + 1, r = nums.length - 1;
    while (l < r) {
      const s = nums[i] + nums[l] + nums[r];
      if (s === 0) { res.push([nums[i], nums[l], nums[r]]); l++; r--; }
      else if (s < 0) l++;
      else r--;
    }
  }
  return res;
}

// 滑动窗口：无重复最长子串
function lengthOfLongestSubstring(s) {
  const set = new Set();
  let l = 0, max = 0;
  for (let r = 0; r < s.length; r++) {
    while (set.has(s[r])) set.delete(s[l++]);
    set.add(s[r]);
    max = Math.max(max, r - l + 1);
  }
  return max;
}
```

### H13. DFS 回溯 — 排列/组合

```javascript
function permute(nums) {
  const res = [], used = Array(nums.length).fill(false);
  function bt(path) {
    if (path.length === nums.length) { res.push([...path]); return; }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) continue;
      used[i] = true;
      path.push(nums[i]);
      bt(path);
      path.pop();
      used[i] = false;
    }
  }
  bt([]);
  return res;
}
```

口诀：选了标记 → 往下走 → 回来撤销标记。

### H14. BFS 队列 + 快速排序

```javascript
// BFS 层序遍历
function levelOrder(root) {
  if (!root) return [];
  const q = [root], res = [];
  while (q.length) {
    const lv = [], sz = q.length;
    for (let i = 0; i < sz; i++) {
      const n = q.shift();
      lv.push(n.val);
      if (n.left) q.push(n.left);
      if (n.right) q.push(n.right);
    }
    res.push(lv);
  }
  return res;
}

// 原地快排
function quickSort(arr, lo = 0, hi = arr.length - 1) {
  if (lo >= hi) return;
  let i = lo, j = hi, pivot = arr[lo];
  while (i < j) {
    while (i < j && arr[j] >= pivot) j--;
    arr[i] = arr[j];
    while (i < j && arr[i] <= pivot) i++;
    arr[j] = arr[i];
  }
  arr[i] = pivot;
  quickSort(arr, lo, i - 1);
  quickSort(arr, i + 1, hi);
}
```

### H15. 链表核心（反转 + 环检测 + 合并）

```javascript
// 反转链表
function reverseList(head) {
  let prev = null, cur = head;
  while (cur) {
    const next = cur.next;
    cur.next = prev;
    prev = cur;
    cur = next;
  }
  return prev;
}

// 环检测（快慢指针）
function hasCycle(head) {
  let slow = head, fast = head;
  while (fast && fast.next) {
    slow = slow.next;
    fast = fast.next.next;
    if (slow === fast) return true;
  }
  return false;
}

// 合并两个有序链表
function mergeTwoLists(l1, l2) {
  const dummy = { next: null };
  let cur = dummy;
  while (l1 && l2) {
    if (l1.val <= l2.val) { cur.next = l1; l1 = l1.next; }
    else { cur.next = l2; l2 = l2.next; }
    cur = cur.next;
  }
  cur.next = l1 || l2;
  return dummy.next;
}
```
