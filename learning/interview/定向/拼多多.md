**1. Promise 串行执行 + 重试机制**

```ts
// 串行执行
async function serialize(tasks: (() => Promise<any>)[]): Promise<any[]> {
  const results = [];
  for (const task of tasks) {
    results.push(await task());
  }
  return results;
}

// 带重试
function retry(fn: () => Promise<any>, times: number, delay: number): Promise<any> {
  return new Promise((resolve, reject) => {
    const attempt = (n: number) => {
      fn().then(resolve).catch(err => {
        if (n <= 1) reject(err);
        else setTimeout(() => attempt(n - 1), delay);
      });
    };
    attempt(times);
  });
}
```

**2. 模板编译函数（`{{}}` 替换）**

```ts
function compile(template: string, data: Record<string, any>): string {
  return template.replace(/\{\{(\w+(?:\.\w+)*)\}\}/g, (_, path) => {
    return path.split('.').reduce((obj, key) => obj?.[key], data) ?? '';
  });
}
// compile('{{user.name}} is {{age}}', {user: {name: 'Tom'}, age: 25})
// → 'Tom is 25'
```

**3. ES6 class 转 ES5（寄生组合继承）**

```ts
function _inherits(Child, Parent) {
  // 原型链继承
  Child.prototype = Object.create(Parent.prototype);
  Child.prototype.constructor = Child;
  // 静态方法继承
  Object.setPrototypeOf(Child, Parent);
}
```

**4. 异步调度器（带并发限制）**

```ts
class Scheduler {
  private queue: (() => Promise<any>)[] = [];
  private running = 0;
  constructor(private max: number) {}

  add(task: () => Promise<any>): Promise<any> {
    return new Promise((resolve, reject) => {
      const run = () => {
        if (this.running >= this.max) return;
        this.running++;
        task().then(resolve).catch(reject).finally(() => {
          this.running--;
          if (this.queue.length) this.queue.shift()!();
        });
      };
      if (this.running < this.max) run();
      else this.queue.push(run);
    });
  }
}
```

**5. 发布订阅模式（EventEmitter）**

```ts
class EventEmitter {
  private events = new Map<string, Function[]>();

  on(event: string, fn: Function) {
    const list = this.events.get(event) || [];
    list.push(fn);
    this.events.set(event, list);
  }

  off(event: string, fn: Function) {
    const list = this.events.get(event);
    if (list) this.events.set(event, list.filter(f => f !== fn));
  }

  emit(event: string, ...args: any[]) {
    (this.events.get(event) || []).forEach(fn => fn(...args));
  }

  once(event: string, fn: Function) {
    const wrapper = (...args: any[]) => { fn(...args); this.off(event, wrapper); };
    this.on(event, wrapper);
  }
}
```

**6. 深拷贝（含循环引用）**

```ts
function deepClone<T>(obj: T, seen = new WeakMap()): T {
  if (obj === null || typeof obj !== 'object') return obj;
  if (seen.has(obj as any)) return seen.get(obj as any);

  if (obj instanceof Date) return new Date(obj) as any;
  if (obj instanceof RegExp) return new RegExp(obj) as any;

  const clone = Array.isArray(obj) ? [] : {} as any;
  seen.set(obj as any, clone);

  for (const key of Object.keys(obj as any)) {
    clone[key] = deepClone((obj as any)[key], seen);
  }
  return clone;
}
```

**7. 防抖 / 节流**

```ts
function debounce(fn: Function, delay: number) {
  let timer: any;
  return function(this: any, ...args: any[]) {
    clearTimeout(timer);
    timer = setTimeout(() => fn.apply(this, args), delay);
  };
}

function throttle(fn: Function, interval: number) {
  let last = 0;
  return function(this: any, ...args: any[]) {
    const now = Date.now();
    if (now - last >= interval) {
      last = now;
      fn.apply(this, args);
    }
  };
}
```

**8. 手写 call / apply / bind**

```ts
Function.prototype.myCall = function(ctx: any, ...args: any[]) {
  const fn = Symbol();
  ctx[fn] = this;
  const result = ctx[fn](...args);
  delete ctx[fn];
  return result;
};

Function.prototype.myBind = function(ctx: any, ...args1: any[]) {
  const fn = this;
  const bound = function(...args2: any[]) {
    return fn.apply(this instanceof bound ? this : ctx, [...args1, ...args2]);
  };
  bound.prototype = Object.create(fn.prototype);
  return bound;
};
```
